// How to use the UIMAPerformer

// Main:
// Initialize at program start

    MongoDBHandler mongoDBHandler = new MongoDBHandler();
    UIMAPerformer uima = new UIMAPerformer(mongoDBHandler);


// XMLProtocolParser:
// after parsing a speech:

    // "text" is the name of the variable containing the full speech text
    // "speech" is the name of a speech object
    JCas jcas = uima.getJCas(text);
    String fullCas = uima.getFullCas(jcas);
    List<MongoToken> tokens = uima.getTokens(jcas);
    List<MongoSentence> sentences= uima.getSentences(jcas);
    List<MongoNamedEntity> namedEntities = uima.getNamedEntities(jcas);
    double sentiment = uima.getAverageSentiment(jcas);
    String mainTopic = uima.getMainTopic(jcas);
    mongoDBHandler.insertSpeech(
            speech,
            fullCas,
            tokens,
            sentences,
            namedEntities,
            sentiment,
            mainTopic
    );
